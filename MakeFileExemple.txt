MakeFileExemple:

all:AppName

#Linking command

AppName: bin/conditionFile1.o bin/conditionFile2.o 

	g++ -o bin/AppName bin/conditionFile1.o bin/conditionFile2.o


#Compile commands for all src file

bin/conditionFile1.o: src/conditionFile1.cpp

	g++ -g -Wall -Weffc++ -std=c++11 -c -Iinclude -o bin/conditionFile1.o src/conditionFile1.cpp	

bin/conditionFile2.o: src/conditionFile2.cpp

	g++ -g -Wall -Weffc++ -std=c++11 -c -Iinclude -o bin/conditionFile2.o src/conditionFile2.cpp	


#Clean the bin directory

clean:
	@echo "cleaning bin"
	rm -f bin/*
